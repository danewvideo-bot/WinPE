Build-All.cmd AVANT:
  ───────────────────────────────────────────────────────────────
  "%PWSH%" -NoProfile -ExecutionPolicy Bypass -STA -File ...
  exit /b %errorlevel%
  ───────────────────────────────────────────────────────────────

Build-All.cmd APRES:
  ───────────────────────────────────────────────────────────────
  echo.
  echo [INFO] Lancement Build-All.ps1...
  echo.
  
  "%PWSH%" -NoProfile -ExecutionPolicy Bypass -STA -File ...
  set "BUILD_ERRORLEVEL=%errorlevel%"
  
  echo.
  echo [INFO] Build termine avec code d'erreur: %BUILD_ERRORLEVEL%
  echo [INFO] Log complet: %LOG%
  echo.
  pause
  
  exit /b %BUILD_ERRORLEVEL%
  ───────────────────────────────────────────────────────────────
